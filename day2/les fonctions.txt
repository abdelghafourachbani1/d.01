Challenge 1 : Fonction de Somme


#include <stdio.h>

int somme(int x,int y) {
    return x+y;
}

int main() {
 
   int a ,b;
 
   printf("entre les deux nombres : \n");
   scanf("%d %d",&a,&b);
   
   printf("la somme des deux valeurs est : %d",somme(a,b));
 
 return 0;
}

Challenge 5 : Fonction de factoriel


#include <stdio.h>

int fct(int x) {
    int i ,fct=1;
    for (i = 1;i <= x;i++) {
        fct *= i;
    }
    return fct;
}

int main() {
 
   int a ;
 
   printf("entre la valeur du nombre : \n");
   scanf("%d",&a);
   
   printf("le factoriel de ce  valeurs est : %d",fct(a));
 
 return 0;
}

chalenge 3 : somme des elements 
#include <stdio.h>

int main() {
    int n, i, s = 0;

    printf("Entrez le nombre d'elements du tableau : ");
    scanf("%d", &n);

    int t[n]; 

    for(i = 0; i < n; i++) {
        printf("Entrez l'element %d : \n", i + 1);
        scanf("%d", &t[i]);
        s += t[i]; 
    }


    printf("La somme des elements du tableau est : %d\n", s);

    return 0;
}

chalenge 4 : le minimum du tableau 
#include <stdio.h>

int main() {
    int n, i,min;

    printf("Entrez le nombre d'elements du tableau : ");
    scanf("%d", &n);

    int t[n]; 

    for(i = 0; i < n; i++) {
        printf("Entrez l'element %d : \n", i + 1);
        scanf("%d", &t[i]);
    }
    
    min = t[0];
    
    for(i = 1; i < n; i++) {
        if (min < t[i]) {
            min = t[i];
        }
    } 


    printf("Le minimum valeurs entre les elements du tableau est : %d\n", min);

    return 0;
}

#include <stdio.h>

int main() {
    int n, i, facteur;

    printf("Nombre d'elements : ");
    scanf("%d", &n);
    int tableau[n];

    for(i = 0; i < n; i++){
        printf("Element %d : ", i+1);
        scanf("%d", &tableau[i]);
    }

    printf("Entrez le facteur de multiplication : ");
    scanf("%d", &facteur);

    printf("Tableau multiplie : ");
    for(i = 0; i < n; i++){

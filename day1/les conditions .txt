### Challenge 4 : Equation du Deuxième Degré

#include <stdio.h>
#include <math.h>

int main() {
    float a, b, c, d, x1, x2;

    printf("Entrez le coefficient a : ");
    scanf("%f", &a);
    printf("Entrez le coefficient b : ");
    scanf("%f", &b);
    printf("Entrez le coefficient c : ");
    scanf("%f", &c);


   
    delta = pow(b ,2) - (4 * a * c);

    if(delta < 0){
        printf("L'équation n'a pas de solution .\n");
    } else if(delta == 0){
        x1 = -b / (2*a);
        printf("L'équation a une solution  : x = %.2f\n", x1);
    } else {
        x1 = (-b - sqrt(delta)) / (2*a);
        x2 = (-b + sqrt(delta)) / (2*a);
        printf("L'équation a deux solution reel : x1 = %.2f , x2 = %.2f\n", x1, x2);
    }

    return 0;
}

Challenge 2 : Calcul de Prime d'Assurance Auto

#include <stdio.h>

int main() {
    int age, type_voiture, accidents;
    float prime_base = 100.0;
    float prime;

    printf("Entrez l'age du conducteur : ");
    scanf("%d", &age);
    printf("Type de voiture (1=Sportive, 2=Utilitaire, 3=Familiale) : ");
    scanf("%d", &type_voiture);
    printf("Nombre d'accidents au cours des 5 dernieres annees : ");
    scanf("%d", &accidents);

    if (age < 25) {
        prime = prime_base * 1.5;
    } else if (age <= 65) {
        prime = prime_base;
    } else {
        prime = prime_base * 1.2;
    }

    switch(type_voiture) {
        case 1: prime *= 2; break;      
        case 2: prime *= 1.2; break;  
        case 3: prime *= 1.1; break;   
        default: printf("Type de voiture invalide!\n"); break;
    }

    
    if (accidents > 1) {
        prime *= 1.3;
    }

    printf("La prime d'assurance est : %.2f euros\n", prime);

    return 0;
}

### conversion de la temperature :###
// Online C compiler to run C program online
#include <stdio.h>

int main(){

 float k;
 int c;
 
   printf("entrer la temperature en celsius : \n");
   scanf("%d",&c);
   
   k = c + 273.15;
   
   printf("la temperature en kelvin : %.2f \n",k);

    return 0;
}

Challenge 1 : Table de Multiplication

#include <stdio.h>

int main() {
    int n, i;

    printf("Entrez un nombre : ");
    scanf("%d", &n);

    printf("Table de multiplication de %d :\n", n);

    for(i = 1; i <= 10; i++) {
        printf("%d * %d = %d\n", n, i, n * i);
    }

    return 0;
}

Challenge 2 : Factorielle d'un Nombre

#include <stdio.h>

int main() {
    
   int x , i , f = 1;
   
   printf("entrer un entier : \n");
   scanf("%d",&x);
   
   for (i = 1; i <= x;i++) {
       
       f *= i;
   }
   
   printf("le factoriel de %d est egale : %d",x,f);

    return 0;
}

Challenge 7 : Inversion d'un Entier

#include <stdio.h>

int main() {
    
   int x , i , inv = 0;
   
   printf("entrer un entier : \n");
   scanf("%d",&x);
   
   do {
       i = x % 10;
       inv = inv * 10 + i;
       x = x / 10;
   } while (x != 0);

   printf("l'inverse de %d est : %d",x,inv);

    return 0;
}

Challenge 9 : Compteur de Chiffres

#include <stdio.h>

int main() {
    
   int x ,org, i , cnt = 0;

   scanf("%d",&x);
   
   org = x;
   
   do {
       i = x % 10;
       x = x / 10;

   } while (x != 0);

   printf("le nombre des chiffres de %d est : %d",org,cnt);

Challenge 9 : Calcul de la distance entre deux points dans un espace 3D

#include <stdio.h>
#include <math.h>   

int main() {
    float x1, y1, z1;
    float x2, y2, z2;
    float d;

    printf("Entrer x1, y1, z1 : ");
    scanf("%f %f %f", &x1, &y1, &z1);

    printf("Entrer x2, y2, z2 : ");
    scanf("%f %f %f", &x2, &y2, &z2);

    d = sqrt(pow(x2 - x1, 2) + pow(y2 - y1, 2) + pow(z2 - z1, 2));

    printf("La distance entre les deux points est : %.4f\n", d);

    return 0;
}

Challenge 12 : Nombre entier à quatre chiffres en ordre inverse

#include <stdio.h>

   int x , i , inv = 0;
   
   printf("entrer un entier : \n");
   scanf("%d",&x);
   
   do {
       i = x % 10;
       inv = inv * 10 + i;
       x = x / 10;

   printf("l'inverse de %d est : %d",x,inv);

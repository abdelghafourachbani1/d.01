#include <stdio.h>
#include <string.h>

int main () {
   
   char c1[10];
   int i ,cnt = 0;
   
   printf("entrer le character : ");
   scanf("%s",c1);
   
   for (i = 0;c1[i] != '\0';i++) {
       cnt++;
   }
   
   printf("le nombre des caracteres est : %d",cnt);
